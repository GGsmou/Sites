import random

col = int(input('col = '))
lin = int(input('lin = '))
persent = int(input('persentage(%) = '))
symb = str(input('symb = '))
symb2 = str(input('symb2 = '))

for i in range(lin):
    for j in range(col):
        if random.randint(1,100) <= persent:
            print(symb, ' ', end = "")
        else:
            print(symb2, ' ', end="")
        j += 1
    print('')
    i += 1
